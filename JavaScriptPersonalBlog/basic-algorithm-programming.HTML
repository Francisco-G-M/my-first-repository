¿Qué es la programación de algoritmos básicos?

La programación de algoritmos básicos se basan en los fundamentos del pensamiento algorítmico,.A su vez estos algoritmos básicos son un conjunto ordenado y determinado de operaciones que permiten la solución de un problema.

convertir Celsius a Fahrenheit

Para convertir Celsius a Fahrenheit usamos la siguiente formula: Celsius múltiplicado por 9/5 + 32:

    
    function convertirTemperaturas(celsius) {
    const FAHRENHEIT = celsius*9/5 + 32;
    return FAHRENHEIT;
    }
    let final = convertirTemperaturas(30);
    console.log(final);

Invertir una cadena

En caso de que quisieramos invertir una cadena podemos hacerlo mediante un ciclo for:

    
    function invertir(string) {
    let newArr = "";
    for(let i = string.length - 1; i >= 0; i--) {
    newArr += string[i]; 
    }
    return newArr;
    }
    let final = invertir("hola");
    console.log(final);

Factorizar un número

Recordemos que un factorial es el producto de todos los enteros menores o iguales al número seleccionado,en este caso vamos a devolver el factorial de 9:

    
    function factorizar(n) {

Encontrar la palabra más larga de una cadena

En caso de que quisieramos encontrar la palabra más larga de una cadena haríamos así:

    
    function theLongWord(str) {
    let longWord = "";
    let word = str.split(" ");
    for(let c = 0; c < word.length; c++) {
    let individualWord = word[c];
    if(individualWord.length > longWord.length) {
    longWord = individualWord;
    }
    }
    return longWord.length;
    }
    let buscar = theLongWord("Cual es la palabra más larga");
    console.log(buscar);
    /*Lo que hacemos es iterar a través de la cadena hasta encontrar la palabra más larga,".split()" lo que hace es separar cada palabra en una
    subcadena dentroo de un arreglo,si ponemos (" ") con un espacio hará lo ya dicho, si no hará lo mismo pero separando y volviendo una subcadena a
    cada letra*/

Devolver los números mayores de los arreglos

Para devolver un arreglo que tenga el número mayor de cada matriz anidad,podemos iterar con un bucle for: 

    
    function ejemplo(array) {
    let newArr = [];
    for(let h = 0; h < array.length; h++) {
    let largestNumber = array[h][0];
    for(let i = 0; i < array[h].length; i++) {
    if(array[h][i] > largestNumber) {
    largestNumber = array[h][i];
    }
    }
    newArr[h] = largestNumber;
    }
    return  newArr;
    }
    var final = ejemplo([[8, 6, 9, 7], [13, 30, 29, 21], [20, 15, 14, 10], [0, 99, 8, 3]]);
    console.log(final);

Confirmar si una cadena termina con el destino dado

Para verificar que una cadena termine con la cadena final dada podemos hacer lo siguiente:

    
    function ending(str, end) {
    let theEnd = end.length;
    let endMatch = str.splice(str.length - theEnd, str.length);
    if(endMatch == end) {
    return true;
    }
    return false;
    }
    var theFinal = ending("Tomas", "s");
    console.log(theFinal); // Esta llamada devuelve true

Repetir una cadena

Para repetir una cadena por una determinada cantidad de veces podemos hacerlo de las siguientes maneras: 

    
    function repeatString(str1, num) {
    let string = "";
    for(let i = 0; i < num; i++) {
    string += str1;
    }
    return string;
    }
    repeatString("W", 3);
    function example2(str2, n) {
    return n > 0 ? str - example2(str2, n - 1) : "";
    }
    console.log(example2(363, 2));
    //La forma más sencilla y práctica para esto sería usando el método .repeat():
    function example3(str3, n2) {
    return str3.repeat(n2);
    }
    console.log(example3("hi", 4));
